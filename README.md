- 💪🏻 Vue3组合式API
- 🔥 基于typeScript

## 下载
```shell
npm install qf-design -S
```

## 快速使用
```javascript
import { createApp } from "vue";
import qfDesign from 'qf-design';
import App from "./app.vue";

const app = createApp(App);
app.use(qfDesign);

app.mount("#app");
```
